294 inline void SdSpi::begin() {
295 #ifdef __AVR_ATmega328P__
296  // Save a few bytes for 328 CPU - gcc optimizes single bit '|' to sbi.
297  PORTB |= 1 << 2; // SS high
298  DDRB |= 1 << 2; // SS output mode
299  DDRB |= 1 << 3; // MOSI output mode
300  DDRB |= 1 << 5; // SCK output mode
301 #else // __AVR_ATmega328P__
302 
303  // set SS high - may be chip select for another SPI device
304  digitalWrite(SS, HIGH);
305 
306  // SS must be in output mode even it is not chip select
307  pinMode(SS, OUTPUT);
308  pinMode(MOSI, OUTPUT);
309  pinMode(SCK, OUTPUT);
310 #endif // __AVR_ATmega328P__
311 }

313 inline void SdSpi::init(uint8_t divisor) {
314  uint8_t b = 2;
315  uint8_t r = 0;
316 
317  // See AVR processor documentation.
318  for (; divisor > b && r < 7; b <<= 1, r += r < 5 ? 1 : 2) {}
319  SPCR = (1 << SPE) | (1 << MSTR) | (r >> 1);
320  SPSR = r & 1 ? 0 : 1 << SPI2X;
321 }

324  SPDR = 0XFF;

326  return SPDR;

329 inline uint8_t SdSpi::receive(uint8_t* buf, size_t n) {
330  if (n-- == 0) {
331  return 0;
332  }
333  SPDR = 0XFF;
334  for (size_t i = 0; i < n; i++) {
335  while (!(SPSR & (1 << SPIF))) {}
336  uint8_t b = SPDR;
337  SPDR = 0XFF;
338  buf[i] = b;
339  }
340  while (!(SPSR & (1 << SPIF))) {}
341  buf[n] = SPDR;
342  return 0;
343 }
344 //------------------------------------------------------------------------------
345 inline void SdSpi::send(uint8_t data) {
346  SPDR = data;
347  while (!(SPSR & (1 << SPIF))) {}
348 }
349 //------------------------------------------------------------------------------
350 inline void SdSpi::send(const uint8_t* buf , size_t n) {
351  if (n == 0) {
352  return;
353  }
354  SPDR = buf[0];
355  if (n > 1) {
356  uint8_t b = buf[1];
357  size_t i = 2;
358  while (1) {
359  while (!(SPSR & (1 << SPIF))) {}
360  SPDR = b;
361  if (i == n) {
362  break;
363  }
364  b = buf[i++];
365  }
366  }
367  while (!(SPSR & (1 << SPIF))) {}
368 }
