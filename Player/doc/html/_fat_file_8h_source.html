426  static bool legal83Char(uint8_t c) {
427  if (c == '"' || c == '|') {
428  return false;
429  }
430  // *+,./
431  if (0X2A <= c && c <= 0X2F && c != 0X2D) {
432  return false;
433  }
434  // :;<=>?
435  if (0X3A <= c && c <= 0X3F) {
436  return false;
437  }
438  // [\]
439  if (0X5B <= c && c <= 0X5D) {
440  return false;
441  }
442  return 0X20 < c && c < 0X7F;
443  }
