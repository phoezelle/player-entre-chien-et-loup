32 static uint8_t CRC7(const uint8_t* data, uint8_t n) {
33  uint8_t crc = 0;
34  for (uint8_t i = 0; i < n; i++) {
35  uint8_t d = data[i];
36  for (uint8_t j = 0; j < 8; j++) {
37  crc <<= 1;
38  if ((d & 0x80) ^ (crc & 0x80)) {
39  crc ^= 0x09;
40  }
41  d <<= 1;
42  }
43  }
44  return (crc << 1) | 1;
45 }

119 bool SdSpiCard::begin(m_spi_t* spi, uint8_t chipSelectPin, uint8_t sckDivisor) {
120  m_errorCode = m_type = 0;
121  m_spi = spi;
122  m_chipSelectPin = chipSelectPin;
123  // 16-bit init start time allows over a minute
124  uint16_t t0 = (uint16_t)millis();
125  uint32_t arg;
126 
127  pinMode(m_chipSelectPin, OUTPUT);
128  digitalWrite(m_chipSelectPin, HIGH);
129  spiBegin();
130 
131  // set SCK rate for initialization commands
132  m_sckDivisor = SPI_SCK_INIT_DIVISOR;
133  spiInit(m_sckDivisor);
134 
135  // must supply min of 74 clock cycles with CS high.
136  for (uint8_t i = 0; i < 10; i++) {
137  spiSend(0XFF);
138  }
139 
140  // command to go idle in SPI mode
141  while (cardCommand(CMD0, 0) != R1_IDLE_STATE) {
142  if (((uint16_t)millis() - t0) > SD_INIT_TIMEOUT) {
143  error(SD_CARD_ERROR_CMD0);
144  goto fail;
145  }
146  }
147 #if USE_SD_CRC
148  if (cardCommand(CMD59, 1) != R1_IDLE_STATE) {
149  error(SD_CARD_ERROR_CMD59);
150  goto fail;
151  }
152 #endif // USE_SD_CRC
153  // check SD version
154  while (1) {
155  if (cardCommand(CMD8, 0x1AA) == (R1_ILLEGAL_COMMAND | R1_IDLE_STATE)) {
156  type(SD_CARD_TYPE_SD1);
157  break;
158  }
159  for (uint8_t i = 0; i < 4; i++) {
160  m_status = spiReceive();
161  }
162  if (m_status == 0XAA) {
163  type(SD_CARD_TYPE_SD2);
164  break;
165  }
166  if (((uint16_t)millis() - t0) > SD_INIT_TIMEOUT) {
167  error(SD_CARD_ERROR_CMD8);
168  goto fail;
169  }
170  }
171  // initialize card and send host supports SDHC if SD2
172  arg = type() == SD_CARD_TYPE_SD2 ? 0X40000000 : 0;
173 
174  while (cardAcmd(ACMD41, arg) != R1_READY_STATE) {
175  // check for timeout
176  if (((uint16_t)millis() - t0) > SD_INIT_TIMEOUT) {
177  error(SD_CARD_ERROR_ACMD41);
178  goto fail;
179  }
180  }
181  // if SD2 read OCR register to check for SDHC card
182  if (type() == SD_CARD_TYPE_SD2) {
183  if (cardCommand(CMD58, 0)) {
184  error(SD_CARD_ERROR_CMD58);
185  goto fail;
186  }
187  if ((spiReceive() & 0XC0) == 0XC0) {
188  type(SD_CARD_TYPE_SDHC);
189  }
190  // Discard rest of ocr - contains allowed voltage range.
191  for (uint8_t i = 0; i < 3; i++) {
192  spiReceive();
193  }
194  }
195  chipSelectHigh();
196  m_sckDivisor = sckDivisor;
197  return true;
198 
199 fail:
200  chipSelectHigh();
201  return false;
202 }

204 // send command and return error code. Return zero for OK
205 uint8_t SdSpiCard::cardCommand(uint8_t cmd, uint32_t arg) {
206  // select card
207  chipSelectLow();
208 
209  // wait if busy
210  waitNotBusy(SD_WRITE_TIMEOUT);
211 
212  uint8_t *pa = reinterpret_cast<uint8_t *>(&arg);
213 
214 #if USE_SD_CRC
215  // form message
216  uint8_t d[6] = {cmd | 0X40, pa[3], pa[2], pa[1], pa[0]};
217 
218  // add crc
219  d[5] = CRC7(d, 5);
220 
221  // send message
222  for (uint8_t k = 0; k < 6; k++) {
223  spiSend(d[k]);
224  }
225 #else // USE_SD_CRC
226  // send command
227  spiSend(cmd | 0x40);
228 
229  // send argument
230  for (int8_t i = 3; i >= 0; i--) {
231  spiSend(pa[i]);
232  }
233 
234  // send CRC - correct for CMD0 with arg zero or CMD8 with arg 0X1AA
235  spiSend(cmd == CMD0 ? 0X95 : 0X87);
236 #endif // USE_SD_CRC
237 
238  // skip stuff byte for stop read
239  if (cmd == CMD12) {
240  spiReceive();
241  }
242 
243  // wait for response
244  for (uint8_t i = 0; ((m_status = spiReceive()) & 0X80) && i != 0XFF; i++) {
245  }
246  return m_status;
247 }

249 uint32_t SdSpiCard::cardSize() {
250  csd_t csd;
251  if (!readCSD(&csd)) {
252  return 0;
253  }
254  if (csd.v1.csd_ver == 0) {
255  uint8_t read_bl_len = csd.v1.read_bl_len;
256  uint16_t c_size = (csd.v1.c_size_high << 10)
257  | (csd.v1.c_size_mid << 2) | csd.v1.c_size_low;
258  uint8_t c_size_mult = (csd.v1.c_size_mult_high << 1)
259  | csd.v1.c_size_mult_low;
260  return (uint32_t)(c_size + 1) << (c_size_mult + read_bl_len - 7);
261  } else if (csd.v2.csd_ver == 1) {
262  uint32_t c_size = 0X10000L * csd.v2.c_size_high + 0X100L
263  * (uint32_t)csd.v2.c_size_mid + csd.v2.c_size_low;
264  return (c_size + 1) << 10;
265  } else {
266  error(SD_CARD_ERROR_BAD_CSD);
267  return 0;
268  }
269 }

299 bool SdSpiCard::erase(uint32_t firstBlock, uint32_t lastBlock) {
300  csd_t csd;
301  if (!readCSD(&csd)) {
302  goto fail;
303  }
304  // check for single block erase
305  if (!csd.v1.erase_blk_en) {
306  // erase size mask
307  uint8_t m = (csd.v1.sector_size_high << 1) | csd.v1.sector_size_low;
308  if ((firstBlock & m) != 0 || ((lastBlock + 1) & m) != 0) {
309  // error card can't erase specified area
310  error(SD_CARD_ERROR_ERASE_SINGLE_BLOCK);
311  goto fail;
312  }
313  }
314  if (m_type != SD_CARD_TYPE_SDHC) {
315  firstBlock <<= 9;
316  lastBlock <<= 9;
317  }
318  if (cardCommand(CMD32, firstBlock)
319  || cardCommand(CMD33, lastBlock)
320  || cardCommand(CMD38, 0)) {
321  error(SD_CARD_ERROR_ERASE);
322  goto fail;
323  }
324  if (!waitNotBusy(SD_ERASE_TIMEOUT)) {
325  error(SD_CARD_ERROR_ERASE_TIMEOUT);
326  goto fail;
327  }
328  chipSelectHigh();
329  return true;
330 
331 fail:
332  chipSelectHigh();
333  return false;
334 }

388 bool SdSpiCard::readData(uint8_t* dst, size_t count) {
389 #if USE_SD_CRC
390  uint16_t crc;
391 #endif // USE_SD_CRC
392  // wait for start block token
393  uint16_t t0 = millis();
394  while ((m_status = spiReceive()) == 0XFF) {
395  if (((uint16_t)millis() - t0) > SD_READ_TIMEOUT) {
396  error(SD_CARD_ERROR_READ_TIMEOUT);
397  goto fail;
398  }
399  }
400  if (m_status != DATA_START_BLOCK) {
401  error(SD_CARD_ERROR_READ);
402  goto fail;
403  }
404  // transfer data
405  if ((m_status = spiReceive(dst, count))) {
406  error(SD_CARD_ERROR_SPI_DMA);
407  goto fail;
408  }
409 
410 #if USE_SD_CRC
411  // get crc
412  crc = (spiReceive() << 8) | spiReceive();
413  if (crc != CRC_CCITT(dst, count)) {
414  error(SD_CARD_ERROR_READ_CRC);
415  goto fail;
416  }
417 #else
418  // discard crc
419  spiReceive();
420  spiReceive();
421 #endif // USE_SD_CRC
422  chipSelectHigh();
423  return true;
424 
425 fail:
426  chipSelectHigh();
427  return false;
428 }
