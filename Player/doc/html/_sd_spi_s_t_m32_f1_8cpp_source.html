66 static void spiDmaRX(uint8_t* dst, uint16_t count) {
67 // spi_rx_dma_enable(SPI1);
68  if (count < 1) return;
69  dma_setup_transfer(DMA1, SPI1_DMAC_RX_CH, &SPI1->regs->DR, DMA_SIZE_8BITS,
70  dst, DMA_SIZE_8BITS, (DMA_MINC_MODE | DMA_TRNS_CMPLT));
71  dma_set_num_transfers(DMA1, SPI1_DMAC_RX_CH, count); // 2 bytes per pixel
72  SPI_DMA_RX_Active = true;
73  dma_enable(DMA1, SPI1_DMAC_RX_CH);
74 }
75 //------------------------------------------------------------------------------
76 // start TX DMA
77 static void spiDmaTX(const uint8_t* src, uint16_t count) {
78  if (count < 1) return;
79  static uint8_t ff = 0XFF;
80 
81  if (!src) {
82  src = &ff;
83  dma_setup_transfer(DMA1, SPI1_DMAC_TX_CH, &SPI1->regs->DR, DMA_SIZE_8BITS,
84  const_cast<uint8_t*>(src), DMA_SIZE_8BITS,
85  (DMA_FROM_MEM | DMA_TRNS_CMPLT));
86  } else {
87  dma_setup_transfer(DMA1, SPI1_DMAC_TX_CH, &SPI1->regs->DR, DMA_SIZE_8BITS,
88  const_cast<uint8_t*>(src), DMA_SIZE_8BITS,
89  (DMA_MINC_MODE | DMA_FROM_MEM | DMA_TRNS_CMPLT));
90  }
91  dma_set_num_transfers(DMA1, SPI1_DMAC_TX_CH, count); // 2 bytes per pixel
92  SPI_DMA_TX_Active = true;
93  dma_enable(DMA1, SPI1_DMAC_TX_CH);
94 }

128 uint8_t SdSpi::receive(uint8_t* buf, size_t n) {
129  int rtn = 0;
130 
131 #if USE_STM32F1_DMAC
132 
133  spiDmaRX(buf, n);
134  spiDmaTX(0, n);
135 
136  uint32_t m = millis();
137  while (SPI_DMA_RX_Active) {
138  if ((millis() - m) > STM32F1_DMA_TIMEOUT) {
139  dmac_channel_disable(SPI_DMAC_RX_CH);
140  dmac_channel_disable(SPI_DMAC_TX_CH);
141  rtn = 2;
142  break;
143  }
144  }
145 
146 #else // USE_STM32F1_DMAC
147  for (size_t i = 0; i < n; i++) {
148  buf[i] = SPI.transfer(0xFF);
149  }
150 #endif // USE_STM32F1_DMAC
151  return rtn;
152 }
